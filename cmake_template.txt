cmake_minimum_required(VERSION 4.2)
 
project(MacroInsight)
enable_language(C ASM CXX)
set(CMAKE_VERBOSE_MAKEFILE TRUE)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
 
# base system
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR arm)
 
# init
set(CMAKE_MSVC_RUNTIME_LIBRARY "")
set(CMAKE_C_COMPILER clang)
set(CMAKE_CXX_COMPILER clang)
set(CMAKE_ASM_COMPILER clang)
# set(CMAKE_C_COMPILER_TARGET arm-arm-none-eabi)
# set(CMAKE_CXX_COMPILER_TARGET arm-arm-none-eabi)
# set(CMAKE_ASM_COMPILER_TARGET arm-arm-none-eabi)
# set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -mcpu=cortex-r5 -mfpu=none")
# set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -mcpu=cortex-r5 -mfpu=none")
# set(CMAKE_ASM_FLAGS "${CMAKE_ASM_FLAGS} -mcpu=cortex-r5 -mfpu=none")
 
# flag
set(CMAKE_C_FLAGS_RELEASE   "-O2 -g")
set(CMAKE_CXX_FLAGS_RELEASE "-O2 -g")
add_compile_options(
    -mthumb
    -Oz
    -Werror
    -Wall
    -Wundef
    -Wno-unused-function
    -MP
)
add_link_options("/nodefaultlib")
 
# 新增包含路徑
include_directories(
    "${PROJECT_SOURCE_DIR}/src_fw/common"
    "${PROJECT_SOURCE_DIR}/src_fw"
)
 
# 定義宏
add_definitions(
    -DMACRO=TRUE
)
 
# 新增原始碼檔案
file(GLOB_RECURSE SOURCES "src_fw/*.h" "src_fw/*.c" "src_fw/*.s" "src_fw/*.S")
list(FILTER SOURCES EXCLUDE REGEX "src_fw/file/notused/.*")

add_library(${PROJECT_NAME} OBJECT ${SOURCES})
 
# generate project structure
# should re-config when change folder structure, because file only work on config
file(GENERATE
    OUTPUT "${VIA_FILE}"
    CONTENT "\"$<JOIN:$<TARGET_OBJECTS:${PROJECT_NAME}>,\"\n\">\""
)
